117  int __attribute__((weak)) configure_as_adc( int port, int pin ){
118 
119  // enable IOCON & A/D clock
120  LPC_SYSCON->SYSAHBCLKCTRL |= ( 0x01 << 16 ) | ( 0x01 << 13 );
121 
122  // enable A/D power
123  LPC_SYSCON->PDRUNCFG &= ~( 1 << 4 );
124 
125  switch( ( port << 8 ) | pin ){
126  case 0x000B: pin_ad_conf( &LPC_IOCON->R_PIO0_11, 0x02 ); return 0;
127 
128  case 0x0100: pin_ad_conf( &LPC_IOCON->R_PIO1_0, 0x02 ); return 1;

287

312 void HWLIB_WEAK wait_us( int_fast32_t n ){
313  // for the default clock frequency of 12 MHz:
314  // the loop uses 12 cycles
315  __asm volatile(
316  " mov r0, %[reg] \t\n"
317  " b 1f \t\n"
318  " .align 4 \t\n"
319  "1: sub r0, #1 \t\n"
320  " b 2f \t\n"
321  "2: b 3f \t\n"
322  "3: nop \t\n"
323  " nop \t\n"
324  " bgt 1b \t\n"
325  :: [reg] "r" (n) : "r0"
326  );
327 }
