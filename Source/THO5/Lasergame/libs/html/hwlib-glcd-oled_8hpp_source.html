200  for( int y = start.y; y < (start.y + size.y); ++y ){
201  int i = x + ( y / 8 ) * size.x;
202  buffer[ i ] = 0x00;

217 private:
218  void write_implementation( location pos, color col ) override {
219  int a = pos.x + ( pos.y / 8 ) * size.x;
220 
221  if( col == foreground ){
222  buffer[ a ] |= ( 0x01 << (pos.y % 8 ));
223  } else {
224  buffer[ a ] &= ~( 0x01 << ( pos.y % 8 ));
